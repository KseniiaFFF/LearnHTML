  <script>
  function validateForm() {
  const email = document.getElementById('email').value;
  const error = document.getElementById('error');
  if (!email.includes('@')) {
  error.innerText = 'Enter a valid email address!’;
  return false;
  }
  error.innerText = '';
  return true;
  }
  </script>

  <div id="scatter-text" style="font-size: 24px; 
  cursor: pointer;">Hover over me!</div>
        

  <script>
  const scatterText = document.getElementById('scatter-text');

  scatterText.onmouseover = () => {
    scatterText.style.position = 'relative';
    scatterText.innerHTML = scatterText.textContent
      .split('')
      .map(letter => `${letter}`)
      .join('');
  };

  scatterText.onmouseout = () => {
    scatterText.innerHTML = 'Hover over me!';
  };
  </script>

  <div id="counter" style="font-size: 32px; 
  text-align: center;">0</div>
        

  <script>
  const counter = document.getElementById('counter');
  let count = 0;
  const target = 10000;

  function updateCounter() {
    if (count < target) {
      count++;
      counter.innerText = count;
      setTimeout(updateCounter, 20);
    }
  }

  updateCounter();
  </script>
